def foobar n
   (1..n).each do |i|
      x = ''
      x << "Foo" if i % 3 == 0
      x << "Bar" if i % 5 == 0
      puts x.empty? ? i : x
   end
end

Fibonnaci Algorithm

Fibonacci numbers are the numbers in the following integer sequence, called the Fibonacci sequence, and characterized by the fact that every number after the first two is the sum of the two preceding ones. 

def fibonacci(n)
   n <= 1 ? n :  fibonacci( n - 1 ) + fibonacci( n - 2 ) 
end
function orbitalPeriod(objects) {
  const GM = 398600.4418;
  const earthRadius = 6367.4447;
  const PI = Math.PI;
   objects.forEach(function(item) {
    var tmp = Math.round(2 * PI * Math.sqrt(Math.pow(earthRadius + item.avgAlt, 3) / GM));
    delete item.avgAlt;
    item.orbitalPeriod = tmp;
  });
   return objects;
}
